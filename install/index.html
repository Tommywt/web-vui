# 安装

# 从github上下载

git clone git@github.com:Tommywt/web-vui.git
1

# 使用npm安装

npm i web-ui -S
1

# 使用方法

在入口文件main.js中引入

import Vue from 'vue'
import Vui from 'web-vui'
import 'web-vui/lib/web-vui.css'

Vue.use(Vui)

new Vue({
  render: h => h(App),
}).$mount('#app')
